class Retangulo {
    constructor(largura, altura) {
        this.largura = largura;
        this.altura = altura;
    }
    
    calcularArea() {
        return this.largura * this.altura;
    }
    
    calcularPerimetro() {
        return 2 * (this.largura + this.altura);
    }
}

let ret1 = new Retangulo(5, 3);
let ret2 = new Retangulo(10, 4);

console.log(ret1.calcularArea());      // 15
console.log(ret2.calcularPerimetro());  // 28
