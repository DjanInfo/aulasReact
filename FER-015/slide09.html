    constructor(largura, altura) {
        this.largura = largura;
        this.altura = altura;
    }
    
    calcularArea() {
        return this.largura * this.altura;

class Aluno {
    constructor(nome, matricula) {
        this.nome = nome;
        this.matricula = matricula;
        this.notas = [];
    }
    
    adicionarNota(nota) {
        this.notas.push(nota);
    }
    
    calcularMedia() {
        if (this.notas.length === 0) return 0;
        let soma = this.notas.reduce((a, b) => a + b, 0);
        return soma / this.notas.length;
    }
    
    estaAprovado() {
        return this.calcularMedia() >= 7;
    }
}

let aluno = new Aluno("Maria", "2024001");
aluno.adicionarNota(8);
aluno.adicionarNota(7);
aluno.adicionarNota(9);

console.log(aluno.calcularMedia());  // 8
console.log(aluno.estaAprovado());   // true
